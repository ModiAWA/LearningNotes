#define IOS                      \
    ios::sync_with_stdio(false); \
    cin.tie(nullptr);            \
    cout.tie(nullptr);
#define endl '\n'
#define ll long long
#define debug(x) cout << #x << ' ' << x << endl;

const int MAXX = 1e7 + 10;
const int inf = 1e9 + 10;
const ll INF = 1e18 + 10;

void test() {
	int n;cin>>n;
    vector<ll> A(n);
    map<ll, int> freq;
    ll maxx = 0;

    for (int i = 0; i < n; i++) {
        cin >> A[i];
        freq[A[i]]++;
        if (A[i] > maxx) maxx = A[i];
    }

    vector<ll> distinct;
    for (auto& p : freq) distinct.push_back(p.first);
    set<ll> cand;
    cand.insert(maxx);
    for (ll val : distinct) {
        cand.insert(maxx + val);
    }

    vector<ll> result;

    for (ll L : cand) {
        int cL = freq.count(L) ? freq[L] : 0;
        if ((n - cL) % 2 != 0 || n - cL < 0) continue;
        bool valid = true;
        for (auto& p : freq) {
            ll a = p.first;
            int cnt = p.second;
            if (a == L) continue; 
            ll b = L - a;
           if (freq.count(b) == 0) {
                valid = false;
                break;
            }
            if (a < b) {
                if (cnt != freq[b]) {
                    valid = false;
                    break;
                }
            }
            if (a == b && a * 2 == L) {
                if (cnt % 2 != 0) {
                    valid = false;
                    break;
                }
            }
        }
        if (valid) {
            result.push_back(L);
        }
    }

    sort(result.begin(), result.end());
    for (size_t i = 0; i < result.size(); i++) {
        if (i > 0) cout << " ";
        cout << result[i];
    }

# K - Pawn Line
