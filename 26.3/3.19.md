### Problem Statement

Here, we define the number of **dots** in a lowercase English letter as follows:

- If the lowercase English letter is `i` or `j`: 11 dot
- If the lowercase English letter is neither `i` nor `j`: 00 dots

Find the sum of the numbers of dots over all characters in S*S*.

### Constraints

- S*S* is a string consisting of lowercase English letters with length between 11 and 1010, inclusive.

### Input

The input is given from Standard Input in the following format:

```
SS
```

### Output

Print the answer.

// 树状数组（Fenwick Tree/BIT）模板，支持单点修改、区间查询，1-based
// 用法：Fenwick<int> fw(n); fw.add(x, d); fw.sum(x); fw.rangeSum(l, r);
// n: 数组长度，x: 下标(1~n)，d: 增量，l/r: 区间端点(1~n)
template <typename T>
struct Fenwick {
	int n;
	std::vector<T> tree;
	Fenwick(int Size = 0) {
		n = Size;
		tree.assign(n + 1, T{});
	}
	// 单点加：下标x加上d
	void add(int x, const T &d) {
		for (int i = x; i <= n; i += (i & -i))
			tree[i] += d;
	}
	// 前缀和：返回[1,x]的和
	T sum(int x) {
		T ans{};
		for (int i = x; i >= 1; i -= (i & -i))
			ans += tree[i];
		return ans;
	}
	// 区间和：返回[l,r]的和
	T rangeSum(int l, int r) { return sum(r) - sum(l - 1); }
};

void test(){
	int n,q;cin>>n>>q;
	vector<int> a(n+1,0);
	Fenwick<ll> fenw(n+1);
	for(int i = 1; i <=n;i++){
		cin>>a[i];
		fenw.add(i,a[i]);

			if(a[x]!=a[x+1]){
				fenw.add(x,a[x+1]-a[x]);
				fenw.add(x+1,a[x]-a[x+1]);
				swap(a[x],a[x+1]);
			}
		}
		else {
			int l,r;cin>>l>>r;
			cout<<fenw.rangeSum(l, r)<<endl;
		}
	}
}

您将获得 Q*Q* 个单词 w1,w2,…,wQ*w*1​,*w*2​,…,*w**Q*​ ，这些单词均来自 AtCoder 所在国家的官方语言。请根据每个单词中包含的字符，判断以下哪项适用：

	经证实，它是高桥语中的一个词。
