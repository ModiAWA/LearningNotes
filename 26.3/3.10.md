### Constraints

```c++
#include<bits/stdc++.h>
using namespace std;

int n;
vector<int> A;
vector<vector<int>> g;
vector<bool> ans;
map<int,int> cnt;
bool duped;

void dfs(int u,int p){
    bool dupe = duped;
    cnt[A[u]]++;
    if(cnt[A[u]] == 2){
        duped = true;
    }
    ans[u] = duped;
    for(int v : g[u]){
        if(v == p) continue;
        dfs(v, u);
    }
    cnt[A[u]]--;
    if(cnt[A[u]] == 1) duped = dupe;
}

void test(){
    cin >> n;
    A.resize(n);
    for(int i = 0; i < n; i++){
        cin >> A[i];
    }
    
    g.assign(n, vector<int>());
    for(int i = 0; i < n-1; i++){
        int u, v;
        cin >> u >> v;
        u--; v--;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    
    ans.resize(n, false);
    duped = false;
    dfs(0, -1); 
    for(int i = 0; i < n; i++) {
        cout << (ans[i] ? "Yes" : "No") << "\n";
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int t = 1;
    while(t--){
        test();
    }
    return 0;
}
```

把三个字母对应位置分到三个数组，用三个指针进行遍历

贪心+三指针

```c++
#include<bits/stdc++.h>
using namespace std;
#define IOS ios::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr);
#define endl '\n' 
#define ll long long
#define debug(x) cout<< #x << ' ' << x << endl;

const int MAXX = 1e7 + 10;
const int inf = 1e9 + 10;
const ll INF = 1e18 + 10;

void test(){
	string s;cin>>s;
	ll len = s.length();
	vector<ll> Apos,Bpos,Cpos;
	for(ll i = 0; i < len;i++){
		if(s[i]=='A') Apos.push_back(i);
		else if(s[i]=='B') Bpos.push_back(i);
		else if(s[i]=='C') Cpos.push_back(i);
	}
	int ia=0,ib=0,ic=0;
	int ans = 0;
	while(ia < Apos.size()&&ib<Bpos.size()&&ic<Cpos.size()){
		while(ib<Bpos.size()&&Bpos[ib]<=Apos[ia]) ib++;
		if(ib>=Bpos.size()) break;
		while (ic < Cpos.size() && Cpos[ic]<= Bpos[ib]) ic++;
		if(ic>=Cpos.size()) break;

		ans++,ia++,ib++,ic++;
	}

	cout<<ans<<endl;

}

int main(){
	IOS
	int t = 1;//cin >> t;
	while(t--){
		test();
	}
	 return 0;
}
```
